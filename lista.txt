    <script>
        document.addEventListener('DOMContentLoaded', function () {
            const registrationsList = document.getElementById('registrationsList');

            // Função para carregar os registros do servidor
            async function loadRegistrations() {
                try {
                    const response = await fetch('http://localhost:3000/registros'); // <-- ATENÇÃO: Verifique a porta do seu servidor
                    if (!response.ok) {
                        throw new Error(`Erro HTTP: ${response.status}`);
                    }
                    const data = await response.json(); // Recebe o array de registros

                    registrationsList.innerHTML = ''; // Limpa a mensagem de carregamento

                    if (data.length === 0) {
                        registrationsList.innerHTML = '<li style="justify-content: center; color: #ccc;">Nenhum registro encontrado.</li>';
                        return;
                    }

                    data.forEach(user => {
                        const listItem = document.createElement('li');
                        listItem.innerHTML = `
                            <span>${user.fullName}</span>
                            <span>${user.age} anos</span>
                        `;
                        registrationsList.appendChild(listItem);
                    });

                } catch (error) {
                    console.error('Erro ao carregar registros:', error);
                    registrationsList.innerHTML = `<li style="color: red; justify-content: center;">Erro ao carregar registros: ${error.message}</li>`;
                }
            }

            // Carrega os registros quando a página é carregada
            loadRegistrations();
        });
    </script>
